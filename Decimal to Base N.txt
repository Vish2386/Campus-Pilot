import java.util.Scanner;

class DecimalToBaseN {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the Input decimal number: ");
        double decimalNumber = sc.nextDouble();
        System.out.print("Enter the base of the destination number system (2-9): ");
        int base = sc.nextInt();
        if (base < 2 || base > 9) {
            System.out.println("Base must be between 2 and 9.");
            return;
        }
        int intPart = (int) decimalNumber;
        double fracPart = decimalNumber - intPart;
        String intResult = convertInteger(intPart, base);
        String fracResult = convertFraction(fracPart, base, 6);
        System.out.println("Decimal " + intPart + " converted into Base-" + base + " system = " + intResult);
        System.out.println("Fractional decimal " + fracPart + " converted into Base-" + base + " system = 0." + fracResult);
        System.out.println("Base-" + base + " equivalent of input decimal = " + intResult + "." + fracResult);
        sc.close();
    }

    private static String convertInteger(int num, int base) {
        if (num == 0) return "0";
        StringBuilder sb = new StringBuilder();
        while (num > 0) {
            int rem = num % base;
            sb.append(rem);
            num /= base;
        }
        return sb.reverse().toString();
    }

    private static String convertFraction(double frac, int base, int precision) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < precision; i++) {
            frac *= base;
            int digit = (int) frac;
            sb.append(digit);
            frac -= digit;
        }
        return sb.toString();
    }
}
